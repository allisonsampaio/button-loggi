## Install

```bash
npm install --save button-loggi
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'button-loggi'
import 'button-loggi/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
